git rev-list --all --objects | \
  awk -v git_dir="$(git rev-parse --git-dir)" \
    '/^[0-9a-f]{40} / {print $2}' | \
  xargs -I {} du -k -d 0 "$git_dir/objects/{}" | \
  sort -n
